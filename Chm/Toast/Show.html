Toast toast; private void showToast(String text) { if (mToast == null) { mToast = Toast.makeText(mContext, text, Toast.LENGTH_SHORT); } else { mToast.setText(text); mToast.setDuration(Toast.LENGTH_SHORT); } mToast.show(); } private void cancelToast() { if (mToast != null) { mToast.cancel(); } } @Override public void onBackPressed() { cancelToast(); super.onBackPressed(); }
