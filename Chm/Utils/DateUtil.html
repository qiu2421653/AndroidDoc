int month = mCalendar.get(Calendar.MONTH); int day = mCalendar.get(Calendar.DAY_OF_MONTH); String mMonth = null; String mDay = null; if (month < 9) mMonth = "0" + (month + 1); else mMonth = String.valueOf(month + 1);
