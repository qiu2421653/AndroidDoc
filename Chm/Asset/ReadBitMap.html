读取图片 

public static Bitmap getImageFromAssets(Context context) { Bitmap image = null; AssetManager am = context.getResources().getAssets(); try { //AndroidStudio方式 InputStream is = getAssets().open("pic.jpg"); InputStream is = getClassLoader().getResourceAsStream("assets/pic.jpg") //Eclipse方式 InputStream is = am.open("file:///android_asset/pic.jpg"); image = BitmapFactory.decodeStream(is); is.close(); } catch (IOException e) { e.printStackTrace(); } return image; }
